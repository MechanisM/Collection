  8  /**

 15 var nimble = (function () {
 16 	"use strict";
 17 	
 18 	return {
 19 		/**
 20 		 * framework name

 22 		 * @constant
 23 		 * @type String
 24 		 */
 25 		name: "nimble",
 26 		/**
 27 		 * framework version
 28 		 * 
 29 		 * @constant
 30 		 * @type String

 33 		/**
 34 		 * return string: framework name + framework version
 35 		 *
 36 		 * @this {nimble}
 37 		 * @return {String}
 38 		 */
 39 		nimble: function () { return this.name + " " + this.version; },
 40 		
 41 		// constants
 42 		CONTEXT_SEPARATOR: " ",
 43 		QUERY_SEPARATOR: "/",
 44 		SUBQUERY_SEPARATOR: "{",
 45 		METHOD_SEPARATOR: "->",
 46 		
 47 		CHILDREN: ">",
 48 		ORDER: ["eq(", ")"],
 49 		//
 50 		
 51 		/**
 52 		 * trim
 53 		 *
 54 		 * @param {String} str
 55 		 * @return {String}
 56 		 */
 57 		trim: function (str) {
 58 			var
 59 				str = str.replace(/^\s\s*/, ''),
 60 				ws = /\s/,
 61 				i = str.length;
 62 			//
 63 			while (ws.test(str.charAt((i -= 1)))){};
 64 			return str.substring(0, i + 1);
 65 		},
 66 		/**
 67 		 * string test
 68 		 *
 69 		 * @param {mixed} obj
 70 		 * @return {Boolean}
 71 		 */
 72 		isString: function (obj) { return Object.prototype.toString.call(obj) === "[object String]"; },
 73 		/**
 74 		 * number test
 75 		 *
 76 		 * @param {mixed} obj
 77 		 * @return {Boolean}
 78 		 */
 79 		isNumber: function (obj) { return Object.prototype.toString.call(obj) === "[object Number]"; },
 80 		/**
 81 		 * boolean test
 82 		 *
 83 		 * @param {mixed} obj
 84 		 * @return {Boolean}
 85 		 */
 86 		isArray: function (obj) { return Object.prototype.toString.call(obj) === "[object Array]"; },
 87 		/**
 88 		 * null && undefined && empty string test
 89 		 *
 90 		 * @param {mixed} obj
 91 		 * @return {Boolean}
 92 		 */
 93 		isExists: function (obj) { return obj !== undefined && obj !== "undefined" && obj !== null && obj !== ""; },
 94 		
 95 		/**
 96 		 * calculate math expression
 97 		 * 
 98 		 * @param {mixed} val - new value
 99 		 * @param {mixed} old - old value
100 		 * @return {mixed}
101 		 */
102 		expr: function (val, old) {
103 			old = old !== undefined || old !== null ? old : "";
104 			//if (this.isString(val) && val.search(/^[+-\\*/]{1}=/) !== -1) {
105 				val = val.split("=");
106 				if (!isNaN(val[1])) { val[1] = +val[1]; }
107 				// simple math
108 				switch (val[0]) {
109 					case "+": { val = old + val[1]; } break;
110 					case "-": { val = old - val[1]; } break;
111 					case "*": { val = old * val[1]; } break;
112 					case "/": { val = old / val[1]; } break;
113 				}
114 			//}
115 		
116 			return val;
117 		},
118 		
119 		/**
120 		 * set new value to object by link or get object by link
121 		 * 
122 		 * @this {nimble}
123 		 * @param {Object|Number|Boolean} obj - some object
124 		 * @param {Context} context - link
125 		 * @param {mixed} [value] - some value
126 		 * @param {Boolean} [del=false] - if "true", remove source element
127 		 * @return {nimble|Boolean|mixed}
128 		 */
129 		byLink: function (obj, context, value, del) {
130 			context = context
131 						.toString()
132 						.replace(new RegExp("\\s*" + this.CHILDREN + "\\s*", "g"), " " + this.CHILDREN + " ")
133 						.split(this.CONTEXT_SEPARATOR);
134 			del = del || false;
135 			
136 			//
137 			var
138 				type = this.CHILDREN,
139 				last = 0, total = 0,
140 				
141 				key, i = context.length,
142 				pos, n,
143 		
144 				objLength, cLength;
145 		
146 			// remove "dead" elements
147 			while ((i -= 1) > -1) {
148 				context[i] = this.trim(context[i]);
149 				if (context[i] === "") {
150 					context.splice(i, 1);
151 					last -= 1;
152 				} else if (context[i] !== this.CHILDREN) {
153 					if (i > last) { last = i; }
154 					total += 1;
155 				}
156 			}
157 			// recalculate length
158 			cLength = context.length;
159 			
160 			// overload
161 			if (obj === false) {
162 				return context.join("");
163 			} else if (this.isNumber(obj)) {
164 				if ((obj = +obj) < 0) { obj += total; }
165 				if (value === undefined) { 
166 					for (i = -1, n = 0; (i += 1) < cLength;) {
167 						if (context[i] !== this.CHILDREN) {
168 							if ((n += 1) === obj) {
169 								context.splice(i + 1, cLength);
170 								return context.join("");
171 							}
172 						}
173 					}
174 				} else {
175 					for (i = cLength, n = 0; (i -= 1) > -1;) {
176 						if (context[i] !== this.CHILDREN) {
177 							if ((n += 1) === obj) {
178 								context.splice(0, i);
179 								return context.join("");
180 							}
181 						}
182 					}
183 				}
184 			}
185 			//
186 			for (i = -1; (i += 1) < cLength;) {
187 				switch (context[i]) {
188 					case this.CHILDREN : { type = context[i]; } break;
189 					default : {
190 						if (type === this.CHILDREN && context[i].substring(0, this.ORDER[0].length) !== this.ORDER[0]) {
191 							if (i === last && value !== undefined) {
192 								if (del === false) {
193 									obj[context[i]] = this.expr(value, obj[context[i]]);
194 								} else {
195 									if (nimble.isArray(obj)) {
196 										obj.splice(context[i], 1);
197 									} else { delete obj[context[i]]; }
198 								}
199 							} else { obj = obj[context[i]]; }
200 						} else {
201 							pos = context[i].substring(this.ORDER[0].length);
202 							pos = pos.substring(0, (pos.length - 1));
203 							pos = +pos;
204 							//
205 							if (this.isArray(obj)) {
206 								if (i === last && value !== undefined) {
207 									if (pos >= 0) {
208 										if (del === false) {
209 											obj[pos] = this.expr(value, obj[pos]);
210 										} else { obj.splice(pos, 1); }
211 									} else {
212 										if (del === false) {
213 											obj[obj.length + pos] = this.expr(value, obj[obj.length + pos]);
214 										} else { obj.splice(obj.length + pos, 1); }
215 									}
216 								} else {
217 									if (pos >= 0) {
218 										obj = obj[pos];
219 									} else { obj = obj[obj.length + pos]; }
220 								}
221 							} else {
222 								if (pos < 0) {
223 									objLength = 0;
224 									for (key in obj) {
225 										if (obj.hasOwnProperty(key)) { objLength += 1; }
226 									}
227 									//
228 									pos += objLength;
229 								}
230 				
231 								n = 0;
232 								for (key in obj) {
233 									if (obj.hasOwnProperty(key)) {
234 										if (pos === n) {
235 											if (i === last && value !== undefined) {
236 												if (del === false) {
237 													obj[key] = this.expr(value, obj[key]);
238 												} else { delete obj[key]; }
239 											} else { obj = obj[key]; }
240 											break;
241 										}
242 										n += 1;
243 									}
244 								}
245 							}
246 						}
247 					}
248 				}
249 			}
250 			
251 			if (value !== undefined) { return this; }
252 			return obj;
253 		},
254 		
255 		/**
256 		 * execute event
257 		 * 
258 		 * @this {nimble}
259 		 * @param {String} query - query string
260 		 * @param {Object} event - event request
261 		 * @param {mixed} [param] - input parameters
262 		 * @param {mixed} [_this=event] - this object
263 		 * @return {mixed}
264 		 */
265 		execEvent: function (query, event, param, _this) {
266 			query = query.split(this.QUERY_SEPARATOR);
267 			param = this.isExists(param) ? param : [];
268 			param = this.isArray(param) ? param : [param];
269 			//
270 			var 
271 				i = -1,
272 				qLength = query.length - 1,
273 				spliter;
274 		
275 			while ((i += 1) < qLength) { event = event[query[i]]; }
276 			//
277 			if (query[i].search(this.SUBQUERY_SEPARATOR) !== -1) {
278 				spliter = query[i].split(this.SUBQUERY_SEPARATOR);
279 				event = event[spliter[0]];
280 				spliter.splice(0, 1);
281 				param = param.concat(spliter);
282 				return event.apply(_this || event, param);
283 			} else { return event[query[i]].apply(_this || event, param); }
284 		},
285 		
286 		/**
287 		 * add new element to object
288 		 *
289 		 * @this {nimble}
290 		 * @param {Plain Object} obj - some object
291 		 * @param {String} active - property name (can use "->unshift" - the result will be similar to work for an array "unshift")
292 		 * @param {mixed} value - some value
293 		 * @return {Plain Object|Boolean}
294 		 */
295 		addElementToObject: function (obj, active, value) {
296 			active = active.split(this.METHOD_SEPARATOR);
297 			var key, newObj = {};
298 		
299 			if (active[1] && active[1] == "unshift") {
300 				newObj[!isNaN(Number(active[0])) ? 0 : active[0]] = value;
301 				for (key in obj) {
302 					if (obj.hasOwnProperty(key)) { newObj[!isNaN(Number(key)) ? +key + 1 : key] = obj[key]; }
303 				}
304 				obj = newObj;
305 		
306 				return obj;
307 			} else if (!active[1] || active[1] == "push") { obj[active[0]] = value; }
308 		
309 			return true;
310 		}
311 	};
